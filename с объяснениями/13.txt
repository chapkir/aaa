using System;

class Program
{
    static void Main()
    {
        Console.Write("Введите целое число m: ");
        int m = int.Parse(Console.ReadLine());

        Console.Write("Введите целое число n: ");
        int n = int.Parse(Console.ReadLine());

        if (m < n)
        {
            PrintAscending(m, n);
        }
        else
        {
            PrintDescending(m, n);
        }
    }

    static void PrintAscending(int start, int end)
    {
        Console.WriteLine($"Числа от {start} до {end} в порядке возрастания:");

        for (int i = start; i <= end; i++)
        {
            Console.Write(i + " ");
        }

        Console.WriteLine();
    }

    static void PrintDescending(int start, int end)
    {
        Console.WriteLine($"Числа от {start} до {end} в порядке убывания:");

        for (int i = start; i >= end; i--)
        {
            Console.Write(i + " ");
        }

        Console.WriteLine();
    }
}


Данный код представляет собой программу на языке программирования C#, которая позволяет пользователю ввести два целых числа m и n и выводит числа от m до n либо в порядке возрастания, либо в порядке убывания, в зависимости от их значений. Вот пошаговое объяснение кода:

В начале программы подключается пространство имен System.

Объявляется класс Program.

В методе Main() происходит основная логика программы. Сначала пользователю выводится приглашение ввести целое число m с помощью Console.Write(). Затем считывается введенное пользователем значение с помощью Console.ReadLine() и преобразуется в целочисленное значение с помощью int.Parse(). Результат сохраняется в переменную m.

Аналогично, пользователю выводится приглашение ввести целое число n и считывается введенное значение. Результат также сохраняется в переменную n.

Далее происходит проверка условия m < n. Если это условие выполняется (число m меньше числа n), то вызывается метод PrintAscending(m, n).

В методе PrintAscending() выводится сообщение о числах от start до end в порядке возрастания с помощью Console.WriteLine(). Затем с помощью цикла for перебираются числа от start до end включительно, и каждое число выводится на консоль с помощью Console.Write(). В конце метода выводится пустая строка с помощью Console.WriteLine() для разделения вывода.

Если условие m < n не выполняется (число m больше или равно числу n), то вызывается метод PrintDescending(m, n).

В методе PrintDescending() выводится сообщение о числах от start до end в порядке убывания с помощью Console.WriteLine(). Затем с помощью цикла for перебираются числа от start до end включительно с шагом -1 (уменьшение на 1 на каждой итерации), и каждое число выводится на консоль с помощью Console.Write(). В конце метода выводится пустая строка с помощью Console.WriteLine() для разделения вывода.

Таким образом, программа позволяет пользователю ввести два целых числа m и n и выводит числа от m до n либо в порядке возрастания, либо в порядке убывания, в зависимости от их значений.
