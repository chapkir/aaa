using System;
 
class Program
{
	static void Main()
	{
    	Console.Write("Введите пароль: ");
    	string password = Console.ReadLine();
 
    	if (IsPasswordGood(password))
    	{
        	Console.WriteLine("Пароль надежный!");
    	}
    	else
    	{
        	Console.WriteLine("Пароль не надежный.");
    	}
	}
 
	static bool IsPasswordGood(string password)
	{
    	// Проверка длины пароля
    	if (password.Length < 8)
    	{
        	return false;
    	}
 
    	// Проверка наличия заглавной буквы
    	bool hasUpper = false;
    	foreach (char c in password)
    	{
        	if (char.IsUpper(c))
        	{
            	hasUpper = true;
            	break;
        	}
    	}
 
    	if (!hasUpper)
    	{
        	return false;
    	}
 
    	// Проверка наличия строчной буквы
    	bool hasLower = false;
    	foreach (char c in password)
    	{
        	if (char.IsLower(c))
        	{
            	hasLower = true;
            	break;
        	}
    	}
 
    	if (!hasLower)
    	{
        	return false;
    	}
 
    	// Проверка наличия цифры
    	bool hasDigit = false;
    	foreach (char c in password)
    	{
        	if (char.IsDigit(c))
        	{
            	hasDigit = true;
            	break;
        	}
    	}
 
    	return hasDigit;
	}
}


Данный код представляет собой простую программу на языке программирования C#, которая проверяет надежность введенного пользователем пароля. Вот пошаговое объяснение кода:

В начале программы объявляется класс Program.

В методе Main() происходит основная логика программы. Сначала пользователю выводится приглашение ввести пароль с помощью Console.Write(). Затем считывается введенная строка пароля с помощью Console.ReadLine() и сохраняется в переменную password.

Далее вызывается метод IsPasswordGood(), который принимает в качестве аргумента пароль и возвращает значение типа bool (истина или ложь) в зависимости от надежности пароля.

В методе IsPasswordGood() происходит проверка надежности пароля. Сначала проверяется длина пароля с помощью условия if (password.Length < 8). Если длина пароля меньше 8 символов, то метод возвращает false, что означает, что пароль не надежный.

Затем происходит проверка наличия заглавной буквы в пароле. Для этого используется цикл foreach, который перебирает каждый символ пароля. Если встречается заглавная буква (char.IsUpper(c)), то переменная hasUpper устанавливается в true. Если после цикла значение переменной hasUpper остается false, то метод возвращает false, что означает, что пароль не надежный.

Аналогичные проверки выполняются для наличия строчной буквы и цифры в пароле. Если пароль не содержит строчных букв или цифр, то метод возвращает false.

Если все проверки пройдены успешно, то метод возвращает true, что означает, что пароль надежный.

В методе Main() результат проверки надежности пароля выводится на консоль с помощью Console.WriteLine() в зависимости от возвращенного значения метода IsPasswordGood(). Если пароль надежный, выводится сообщение "Пароль надежный!", иначе выводится сообщение "Пароль не надежный.".

Таким образом, программа проверяет пароль на длину (не менее 8 символов), наличие заглавной буквы, строчной буквы и цифры, и сообщает пользователю, является ли пароль надежным или нет.
